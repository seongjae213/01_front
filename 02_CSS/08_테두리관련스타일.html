테두리 관련 스타일 (border~)

        선택자 {
            /* 테두리 스타일(종류) */
            border[-위치]-style: none(기본값) / solid / dotted / dashed / double / groove / ridge / inset / outset;
            
            /* 테두리 두께 */
            border[-위치]-width: px/em/%;

            /* 테두리 색상 */
            border[-위치]-color: blue;

            /* 위의 항목들 한번에 적용 */
            border[-위치]: 두께 스타일 [색상](작성안하면 검은색);

            /* 테두리 모서리 둥글게 둥글게 */
            border[-상하위치-좌우위치]-radius: 값;
        
            /* 테두리(box) 그림자 효과 */
            box-shadow: 가로거리(x) 세로거리(y) 흐림정도 번짐정도 색상;
        }
